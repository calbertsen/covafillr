158 template<typename scalartype_>
159 typename tricubicInterpolation<scalartype_>::vecmattype tricubicInterpolation<scalartype_>::makeAlpha(covafill<scalartype>* cf,vectortype minCoord,vectortype maxCoord) {
160 
161  matrixtype coords(2,minCoord.size());
162  for(int i = 0; i < minCoord.size(); ++i){
163  coords(0,i) = minCoord(i);
164  coords(1,i) = maxCoord(i);
165  }
166 
167  matrixtype Minv = getMinv();
168 
169  int ncoef = 64;
170 
171  vectortype d = maxCoord - minCoord;
172  vectortype mult(8);
173  mult << 1, d(0), d(1), d(2),
174  d(0) * d(1), d(0) * d(2), d(1) * d(2), d(1) * d(2) * d(3);
175  vectortype x(ncoef);
176  x.setZero();
177 
178  // Order of x:
179  // f at each corner (1-8)
180  // f/dx, df/dy, df/dz,
181  // d2f/dxdy, d2f/dxdz, d2f/dydz
182  // d3f/dxdydz (all of these are assumed 0)
183  vectortype y(4), y1(4), y2(4), y12(4);
184  int indxs[8][3] = {{0,0,0},{1,0,0},
185  {0,1,0},{1,1,0},
186  {0,0,1},{1,0,1},
187  {0,1,1},{1,1,1}};
188 
189  for(int i = 0; i < 8; ++i){
190  vectortype tmp(3);
191  tmp(0) = coords(indxs[i][0],0);
192  tmp(1) = coords(indxs[i][1],1);
193  tmp(2) = coords(indxs[i][2],2);
194  vectortype vals = cf->operator()(tmp, true);
195  int nv = vals.size();
196  x(i) = vals(0)*mult(0); // f(x,y,z)
197  x(i+8) = (nv > 1) ? vals(1)*mult(1) : 0.0; // f_x(x,y,z)
198  x(i+2*8) = (nv > 2) ? vals(2)*mult(2) : 0.0; // f_y(x,y,z)
199  x(i+3*8) = (nv > 3) ? vals(3)*mult(3) : 0.0; // f_z(x,y,z)
200  x(i+4*8) = (nv > 5) ? vals(5)*mult(4) : 0.0; // f_{x,y}(x,y,z)
201  x(i+5*8) = (nv > 6) ? vals(6)*mult(5) : 0.0; // f_{x,z}(x,y,z)
202  x(i+6*8) = (nv > 5) ? vals(8)*mult(6) : 0.0; // f_{y,z}(x,y,z)
203  x(i+7*8) = 0.0; // f_{x,y,z}(x,y,z)
204  }
205 
206 
207  vectortype alphaCoef = Minv * x.matrix();
208  vecmattype coefs(4);
209  for(int i = 0; i < 4; ++i){
210  coefs(i).resize(4,4);
211  coefs(i).setZero();
212  }
213  int indx = 0;
214  // Stacking in Lekien & Marsden 2005 is:
215  // coefs(i)(j,k) = alpha(i+4*j+16*k)
216  for(int k = 0; k < 4; ++k)
217  for(int j = 0; j < 4; ++j)
218  for(int i = 0; i < 4; ++i)
219  coefs(i)(j,k) = alphaCoef(indx++);
220 
221  return coefs;
222 };

237 typename tricubicInterpolation<scalartype_>::vectortype tricubicInterpolation<scalartype_>::operator()(vectortype newcoord) {
238  vectortype newval(4);
239  newval.setZero();
240 
241  vectortype d = this->maxCoord - this->minCoord;
242  vectortype x = newcoord - this->minCoord;
243  scalartype t = x(0)/d(0);
244  scalartype u = x(1)/d(1);
245  scalartype v = x(2)/d(2);
246 
247  if(t > 1 || t < 0 || u > 1 || u < 0 || v > 1 || v < 0){
248  // Trow exception
249  return newval/(scalartype)0.0;
250  }
251 
252 
253  for(int i = 0; i < 4; ++i)
254  for(int j = 0; j < 4; ++j)
255  for(int k = 0; k < 4; ++k){
256  // Function value
257  newval(0) += alpha(i)(j,k) * pow(t,i) * pow(u,j) * pow(v,j);
258  // Derivative of first variable - problem if t==0??
259  if(i > 0)
260  newval(1) += scalartype(i) * alpha(i)(j,k) * pow(t,i-1) * pow(u,j) * pow(v,k) / d(0);
261  // Derivative of second variable
262  if(j > 0)
263  newval(2) += scalartype(j) * alpha(i)(j,k) * pow(t,i) * pow(u,j-1) * pow(v,k) / d(1);
264  // Derivative of third variable
265  if(j > 0)
266  newval(3) += scalartype(k) * alpha(i)(j,k) * pow(t,i) * pow(u,j) * pow(v,k-1) / d(2);
267  }
268  return newval;
