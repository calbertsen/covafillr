90 typename unicubicInterpolation<scalartype_>::matrixtype unicubicInterpolation<scalartype_>::makeAlpha(covafill<scalartype>* cf,vectortype minCoord,vectortype maxCoord) {
91 
92  matrixtype coords(2,minCoord.size());
93  for(int i = 0; i < minCoord.size(); ++i){
94  coords(0,i) = minCoord(i);
95  coords(1,i) = maxCoord(i);
96  }
97 
98  matrixtype Minv = getMinv();
99 
100  int ncoef = 4;
101 
102  vectortype d = maxCoord - minCoord;
103  vectortype mult(2);
104  mult << 1, d(0);
105 
106  vectortype x(ncoef);
107  x.setZero();
108 
109  for(int i = 0; i < coords.rows(); ++i){
110  vectortype tmp(1);
111  tmp(0) = coords(i,0);
112  vectortype vals = cf->operator()(tmp);
113  x.segment(2*i,2) = vals.head(2) * mult;
114  }
115 
116  vectortype alphaCoef = Minv * x.matrix();
117  matrixtype coefs(4,ncoef/4);
118  coefs.setZero();
119  int indx = 0;
120  for(int j = 0; j < ncoef/4; ++j)
121  for(int i = 0; i < 4; ++i)
122  coefs(i,j) = alphaCoef(indx++);
123 
124  return coefs;
125 };

140 typename unicubicInterpolation<scalartype_>::vectortype unicubicInterpolation<scalartype_>::operator()(vectortype newcoord) {
141 
142  vectortype newval(2);
143  newval.setZero();
144 
145  vectortype d = this->maxCoord - this->minCoord;
146  vectortype x = newcoord - this->minCoord;
147  scalartype t = x(0)/d(0);
148 
149  if(t > 1 || t < 0){
150  // Trow exception
151  return newval/(scalartype)0.0;
152  }
153 
154 
155  for(int i = 0; i < 4; ++i){
156  // Function value
157  newval(0) += alpha(i,0) * pow(t,i);
158  // Derivative of first variable - problem if t==0??
159  if(i > 0)
160  newval(1) += i * alpha(i,0) * pow(t,i-1) / d(0);
161  // Derivative of second variable
162  }
163  return newval;
164 };
